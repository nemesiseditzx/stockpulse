# ============================= # 🕌 PREMIUM HALAL SYSTEM # ============================= HALAL_LIST = { "AAPL","MSFT","NVDA","AMD","GOOGL","META","TSLA","AMZN","NFLX" } HARAM_LIST = { "JPM","BAC","C","GS","WFC","MS","AXP" } @app.get("/halal/{symbol}") def halal(symbol: str): sym = symbol.upper().replace("-USD", "") # 1️⃣ Hard rule if sym in HALAL_LIST: return {"status": "HALAL"} if sym in HARAM_LIST: return {"status": "HARAM"} # 2️⃣ Smart sector detection try: stock = yf.Ticker(sym) info = stock.info sector = str(info.get("sector", "")).lower() industry = str(info.get("industry", "")).lower() haram_keywords = [ "bank", "financial", "insurance", "credit", "lending", "capital markets" ] if any(k in sector for k in haram_keywords) or any(k in industry for k in haram_keywords): return {"status": "HARAM"} return {"status": "HALAL"} except: return {"status": "UNKNOWN"}
